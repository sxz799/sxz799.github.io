    #从一行以,分隔的数中读取坐标，放入元组并返回

        try:
            if eval(a[i])>0:

            a[i]=0
    return tuple(a)
def distance(point):
    #计算point与原点的距离并返回，要math库中的函数
    sum=0
    for i in range(3):
        sum=sum+point[i]*point[i]
    return math.sqrt(sum)

            a.append(points[i])
        else:
            b.append(points[i])
    z=(a,b)

    sum=0
    for i in range(len(pointList)):
